Usage:

use your terminal: 

      # list services
      brew services list

      # restart / start / stop services
      brew services restart httpd22

      # edit your php.ini
      open /usr/local/etc/php/5.5/php.ini

      # local apache config
      open /LocalSites/httpd-vhosts.conf

      # log files
      open /LocalSites/logs

      # create new website
      mkdir /LocalSites/FirstWebsite
      echo "
 my first site 
" >> /LocalSites/FirstWebsite/index.html
      open "http://FirstWebsite.dev:8080/"

      # manage databases => I hope you remember the password you set
      open "http://localhost:8080/adminer"
